378  LoomModule* Create(LoomManager* manager, JsonVariant module) const
379  {
380  const char* name = module["name"];
381 
382  for (auto elem : factory::to_array(LookupTable) ) {
383  if ( strcmp(name, elem.name) == 0 ) {
384 
385  if (module["params"].is<JsonArray>()) {
386  // Generate according to list of parameters
387  LPrintln("[Factory] Creating: ", name);
388  return elem.Construct(manager, module["params"]);
389  } else if ( module["params"].is<const char*>() && strcmp(module["params"], "default") == 0 ) {
390  // Generate using default parameters
391  if (elem.ConstructDefault == nullptr) {
392  LPrintln("[Factory] ", "No default constructor for module '", name, "'");
393  return nullptr;
394  } else {
395  return elem.ConstructDefault(manager);
396  }
397  } else {
398  // Invalid parameters
399  LPrintln("[Factory] ", "Invalid parameters in module '", name, "'");
400  return nullptr;
401  }
402  }
403  }
404 
405  LPrintln("'", name, "' could not be created");
406  return nullptr;
407  }
