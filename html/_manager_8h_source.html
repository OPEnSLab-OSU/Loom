311  bool add_data(const char* module, const char* key, const T val)
312  {
313  if ( doc.isNull() ) {
314  doc["type"] = "data";
315  }
316  JsonObject json = doc.as<JsonObject>();
317  if (strcmp(json["type"], "data") == 0 ) {
318  JsonObject data = get_module_data_object(json, module);
319  data[key] = val;
320  // print an error if the JSON is full
321  if (doc.memoryUsage() > doc.capacity() - 10)
322  LPrintln("Warning: JSON document overflowed! Loom does not have enough memory for the number of sensors you are reading, please use less sensors or submit an issue.");
323  return true;
324  } else {
325  return false;
326  }
327  }
